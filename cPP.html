C Programming Questions (53–100)

53. WAP to print your name 5 times using loop.
#include <stdio.h>
int main() {
    for(int i=1; i<=5; i++) {
        printf("Ajay Wadekar\n");
    }
    return 0;
}
54. WAP to print your name 5 times without using loop.
#include <stdio.h>
int main() {
    printf("Ajay Wadekar\nAjay Wadekar\nAjay Wadekar\nAjay Wadekar\nAjay Wadekar\n");
    return 0;
}

58. WAP to print natural numbers that are divisible by 5 from 1 to 100.

61. WAP to print the following series: 1 5 2 4 3 3 4 2 5 1
#include <stdio.h>
int main() {
    int a=1, b=5;
    while(a<=5 && b>=1) {
        printf("%d %d ", a, b);
        a++;
        b--;
    }
    return 0;
}
62. WAP to print Fibonacci series up to 10 terms.
#include <stdio.h>
int main() {
    int n1=0, n2=1, n3;
    printf("%d %d ", n1, n2);
    for(int i=2; i<10; i++) {
        n3 = n1 + n2;
        printf("%d ", n3);
        n1 = n2;
        n2 = n3;
    }
    return 0;
}
